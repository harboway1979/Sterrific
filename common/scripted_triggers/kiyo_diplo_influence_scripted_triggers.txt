kiyo_diplo_influence_has_added_influence = {
	OR = {
		kiyo_diplo_influence_has_valid_diplo_relations = yes
		AND = {
			kiyo_diplo_influence_has_valid_diplo_relations = no
			kiyo_diplo_influence_has_diplo_modifiers = yes
		}
	}
}

# THIS = COUNTRY
kiyo_diplo_influence_has_valid_diplo_relations = {
	any_country = {
		NOT = { is_country = prev }
		OR = {
			has_non_aggression_pact = prev
			has_association_status = prev
			AND = {
				prev = { is_guaranteeing = prev }
				NOT = { is_guaranteeing = prev }
			}
		}
	}
}
