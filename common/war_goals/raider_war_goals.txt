wg_stop_raiding = {
	casus_belli = cb_stop_raiding

	hide_if_no_cb = yes
	cede_claims = no

	set_defender_wargoal = wg_plunder

	surrender_acceptance = -100

	potential = {
		has_total_war_cb = no
	}
	possible = {
		always = yes
	}

	on_status_quo = {
		custom_tooltip = wg_stop_raiding_status_quo
		hidden_effect = {
			every_owned_planet = {
				from = {
					every_owned_pop = {
						limit = {
							has_pop_flag = abducted_pop_@root
							has_pop_flag = abducted_from_@prevprev
							planet = { is_controlled_by = root }
						}
						resettle_pop = { pop = this planet = prevprev }
						remove_pop_flag = abducted_pop_@root
						remove_pop_flag = abducted_from_@prevprev
					}
				}
			}
			# Any pops left move to capital
			from = {
				every_owned_pop = {
					limit = {
						has_pop_flag = abducted_pop_@root
						planet = { is_controlled_by = root }
					}
					resettle_pop = { pop = this planet = root.capital_scope }
					remove_pop_flag = abducted_pop_@root
				}
			}
			if = {
				limit = { from = { NOT = { any_owned_pop = { has_pop_flag = abducted_pop_@root } } } }
				remove_relation_flag = { flag = abducted_pops who = from }
			}
		}
	}

	on_accept = {
		custom_tooltip = wg_stop_raiding_accept
		add_resource = { influence = 50 }
		from = { add_modifier = { modifier = humiliated days = 1800 } }
		hidden_effect = {
			# Free any pops abducted from us.
			every_owned_planet = {
				from = {
					every_owned_pop = {
						limit = {
							has_pop_flag = abducted_pop_@root
							has_pop_flag = abducted_from_@prevprev
						}
						resettle_pop = { pop = this planet = prevprev }
						remove_pop_flag = abducted_pop_@root
						remove_pop_flag = abducted_from_@prevprev
					}
				}
			}
			# Any pops left move to capital
			from = {
				every_owned_pop = {
					limit = {
						has_pop_flag = abducted_pop_@root
					}
					resettle_pop = { pop = this planet = root.capital_scope }
					remove_pop_flag = abducted_pop_@root
				}
			}
			remove_relation_flag = { flag = abducted_pops who = from }
		}
	}

	ai_weight = {
		weight = 100
	}
}
